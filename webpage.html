### Insertion Sort ```python def insertion_sort(nums): for i in range(1, len(nums)): key = nums[i] j = i - 1 while j >= 0 and nums[j] > key: nums[j + 1] = nums[j] j -= 1 nums[j + 1] = key ``` ### Heap Sort ```python def heapify(nums, n, i): largest = i left = 2 * i + 1 right = 2 * i + 2 if left < n and nums[left]> nums[largest]: largest = left if right < n and nums[right]> nums[largest]: largest = right if largest != i: nums[i], nums[largest] = nums[largest], nums[i] heapify(nums, n, largest) def heap_sort(nums): n = len(nums) for i in range(n // 2 - 1, -1, -1): heapify(nums, n, i) for i in range(n - 1, 0, -1): nums[0], nums[i] = nums[i], nums[0] heapify(nums, i, 0) ``` ### Performance Profiling ```python import time import random def profile(sizes): for n in sizes: base = [random.randint(0, 10000) for _ in range(n)] #
